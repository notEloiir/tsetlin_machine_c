487 void tm_initialize(struct TsetlinMachine *tm) {
488  tm->mid_state = (tm->max_state + tm->min_state) / 2;
489  tm->s_inv = 1.0f / tm->s;
490  tm->s_min1_inv = (tm->s - 1.0f) / tm->s;
491 
492  // Initialize clauses (TA states making up the clauses)
493  // pairs of positive and negative literals randomly (-1, 0) or (0, -1) if mid_state is 0
494  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
495  for (uint32_t literal_id = 0; literal_id < tm->num_literals; literal_id++) {
496  if (prng_next_float(&(tm->rng)) <= 0.5) {
497  // positive literal
498  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 0] = tm->mid_state - 1;
499  // negative literal
500  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] = tm->mid_state;
501  } else {
502  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 0] = tm->mid_state;
503  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] = tm->mid_state - 1;
504  }
505  }
506  }
507 
508  // Init weights randomly to -1 or 1
509  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
510  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
511  tm->weights[(clause_id * tm->num_classes) + class_id] = 1 - 2*(prng_next_float(&(tm->rng)) <= 0.5);
512  }
513  }
514 }

533 static inline void calculate_clause_output(struct TsetlinMachine *tm, const uint8_t *X, uint8_t skip_empty) {
534  // For each clause, check if it is "active" - all necessary literals have the right value
535  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
536  tm->clause_output[clause_id] = 1;
537  uint8_t empty_clause = 1;
538 
539  // Clause is active if:
540  // - it's not empty (unless skip_empty is unset as should be the case for training)
541  // - each literal present in the clause has the right value (same as the input X)
542  for (uint32_t literal_id = 0; literal_id < tm->num_literals; literal_id++) {
543  uint8_t action_include = action(tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 0], tm->mid_state);
544  uint8_t action_include_negated = action(tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1], tm->mid_state);
545 
546  empty_clause = (empty_clause && !(action_include || action_include_negated));
547 
548  if ((action_include == 1 && X[literal_id] == 0) || (action_include_negated == 1 && X[literal_id] == 1)) {
549  tm->clause_output[clause_id] = 0;
550  break;
551  }
552  }
553 
554  if (empty_clause && skip_empty) {
555  tm->clause_output[clause_id] = 0;
556  }
557  }
558 }

591 // Type I Feedback
592 // Applied if clause at clause_id voted correctly for class at class_id
593 
594 // Type I a - Clause is active for literals X (clause_output == 1)
595 // Meaning: it's active and voted correctly
596 // Action: reinforce the clause TAs and weights
597 // Intuition: so that it continues to vote for the same class
601 static inline void type_1a_feedback(struct TsetlinMachine *tm, const uint8_t *X, uint32_t clause_id, uint32_t class_id) {
602  // float s_inv = 1.0f / tm->s;
603  // float s_min1_inv = (tm->s - 1.0f) / tm->s;
604 
605  uint8_t feedback_strength = 1;
606 
607  // Reinforce the clause weight (away from mid_state)
608  if (tm->weights[clause_id * tm->num_classes + class_id] >= 0) {
609  tm->weights[clause_id * tm->num_classes + class_id] += min(feedback_strength, SHRT_MAX - tm->weights[clause_id * tm->num_classes + class_id]);
610  }
611  else {
612  tm->weights[clause_id * tm->num_classes + class_id] -= min(feedback_strength, -(SHRT_MIN - tm->weights[clause_id * tm->num_classes + class_id]));
613  }
614 
615  // Reinforce the Tsetlin Automata states
616  for (uint32_t literal_id = 0; literal_id < tm->num_literals; literal_id++) {
617  if (X[literal_id] == 1) {
618  // True positive
619  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)] +=
620  min(tm->max_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)], feedback_strength) * (
621  (tm->boost_true_positive_feedback == 1 || prng_next_float(&(tm->rng)) <= tm->s_min1_inv));
622 
623  // False negative
624  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] -=
625  min(-(tm->min_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1]), feedback_strength) * (
626  (prng_next_float(&(tm->rng)) <= tm->s_inv));
627 
628  } else {
629  // True negative
630  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] +=
631  min(tm->max_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1], feedback_strength) * (
632  (prng_next_float(&(tm->rng)) <= tm->s_min1_inv));
633 
634  // False positive
635  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)] -=
636  min(-(tm->min_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)]), feedback_strength) * (
637  (prng_next_float(&(tm->rng)) <= tm->s_inv));
638  }
639  }
640 }
641 
642 
643 // Type I b - Clause is inactive for literals X (clause_output == 0)
644 // Meaning: it's inactive but would have voted correctly
645 // Action: lower the clause TAs, both positive and negative, towards exclusion
646 // Intuition: so that it "finds something else to do", "countering force"
650 static inline void type_1b_feedback(struct TsetlinMachine *tm, uint32_t clause_id) {
651  // float s_inv = 1.0f / tm->s;
652 
653  uint8_t feedback_strength = 1;
654 
655  // Penalize the clause TAs (towards min_state - exclusion)
656  for (uint32_t literal_id = 0; literal_id < tm->num_literals; literal_id++) {
657  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)] -=
658  min(-(tm->min_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)]), feedback_strength) * (
659  (prng_next_float(&(tm->rng)) <= tm->s_inv));
660 
661  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] -=
662  min(-(tm->min_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1]), feedback_strength) * (
663  (prng_next_float(&(tm->rng)) <= tm->s_inv));
664  }
665 }
666 
667 
668 // Type II Feedback
669 // Clause at clause_id voted incorrectly for class at class_id
670 // && Clause is active for literals X (clause_output == 1)
671 // Meaning: it's active but voted incorrectly
672 // Action: raise excluded clause TAs that could deactivate the clause is included (towards inclusion)
673 // and punish the clause weight (towards zero)
674 // Intuition: either fix the weight or exclude the clause, whichever is easier
678 static inline void type_2_feedback(struct TsetlinMachine *tm, const uint8_t *X, uint32_t clause_id, uint32_t class_id) {
679  uint8_t feedback_strength = 1;
680 
681  tm->weights[clause_id * tm->num_classes + class_id] +=
682  tm->weights[clause_id * tm->num_classes + class_id] >= 0 ? -feedback_strength : feedback_strength;
683 
684  for (uint32_t literal_id = 0; literal_id < tm->num_literals; literal_id++) {
685  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)] +=
686  min(tm->max_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)], feedback_strength) * (
687  0 == action(tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2)], tm->mid_state) &&
688  0 == X[literal_id]);
689 
690  tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1] +=
691  min(tm->max_state - tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1], feedback_strength) * (
692  0 == action(tm->ta_state[(((clause_id * tm->num_literals) + literal_id) * 2) + 1], tm->mid_state) &&
693  1 == X[literal_id]);
694  }
695 }

839 void tm_apply_feedback(struct TsetlinMachine *tm, uint32_t clause_id, uint32_t class_id, uint8_t is_class_positive, const uint8_t *X) {
840  uint8_t is_vote_positive = tm->weights[(clause_id * tm->num_classes) + class_id] >= 0;
841  if (is_vote_positive == is_class_positive) {
842  if (tm->clause_output[clause_id] == 1) {
843  type_1a_feedback(tm, X, clause_id, class_id);
844  }
845  else {
846  type_1b_feedback(tm, clause_id);
847  }
848  }
849  else if (tm->clause_output[clause_id] == 1) {
850  type_2_feedback(tm, X, clause_id, class_id);
851  }
852 }

860 void tm_feedback_class_idx(struct TsetlinMachine *tm, const uint8_t *X, const void *y) {
861  // Pick positive and negative classes based on the label:
862  // Positive class is the one that matches the label,
863  // negative is randomly chosen from the rest, weighted by votes
864  const uint32_t *label_ptr = (const uint32_t *)y;
865  const uint32_t positive_class = *label_ptr;
866  uint32_t negative_class = 0;
867 
868  // Calculate class update probabilities:
869  // Positive class is inversely proportional to the votes for it, (avoiding overfitting)
870  // Negative class is proportional to the votes for it (more sure it should not be chosen)
871  int32_t votes_clipped_positive = clip(tm->votes[positive_class], (int32_t)tm->threshold);
872  float update_probability_positive = ((float)tm->threshold - (float)votes_clipped_positive) / (float)(2 * tm->threshold);
873 
874  // Apply feedback to: chosen classes - every clause
875  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
876  if (prng_next_float(&(tm->rng)) <= update_probability_positive) {
877  tm_apply_feedback(tm, clause_id, positive_class, 1, X);
878  }
879  }
880 
881  // Continue for negative class
882  int32_t sum_votes_clipped_negative = 0;
883  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
884  if (class_id != positive_class) {
885  sum_votes_clipped_negative += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
886  }
887  }
888  if (sum_votes_clipped_negative == 0) return;
889  int32_t random_vote_negative = prng_next_uint32(&(tm->rng)) % sum_votes_clipped_negative;
890  int32_t accumulated_votes = 0;
891  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
892  if (class_id != positive_class) {
893  accumulated_votes += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
894  if (accumulated_votes >= random_vote_negative) {
895  negative_class = class_id;
896  break;
897  }
898  }
899  }
900 
901  int32_t votes_clipped_negative = clip(tm->votes[negative_class], (int32_t)tm->threshold);
902  float update_probability_negative = ((float)votes_clipped_negative + (float)tm->threshold) / (float)(2 * tm->threshold);
903 
904  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
905  if (prng_next_float(&(tm->rng)) <= update_probability_negative) {
906  tm_apply_feedback(tm, clause_id, negative_class, 0, X);
907  }
908  }
909 }

914 void tm_feedback_bin_vector(struct TsetlinMachine *tm, const uint8_t *X, const void *y) {
915  // Pick positive and negative classes based on the label:
916  // Positive is randomly chosen from the ones that matches the label, weighted by votes,
917  // negative is randomly chosen from the rest, weighted by votes
918  const uint8_t *label_arr = (const uint8_t *)y;
919  uint32_t positive_class = 0;
920  uint32_t negative_class = 0;
921 
922  int32_t sum_votes_clipped_positive = 0;
923  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
924  if (label_arr[class_id]) {
925  sum_votes_clipped_positive += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
926  }
927  }
928  if (sum_votes_clipped_positive == 0) goto negative_feedback;
929  int32_t random_vote_positive = prng_next_uint32(&(tm->rng)) % sum_votes_clipped_positive;
930  int32_t accumulated_votes_positive = 0;
931  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
932  if (label_arr[class_id]) {
933  accumulated_votes_positive += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
934  if (accumulated_votes_positive >= random_vote_positive) {
935  positive_class = class_id;
936  break;
937  }
938  }
939  }
940 
941  // Calculate class update probabilities:
942  // Positive class is inversely proportional to the votes for it, (avoiding overfitting)
943  // Negative class is proportional to the votes for it (more sure it should not be chosen)
944  int32_t votes_clipped_positive = clip(tm->votes[negative_class], (int32_t)tm->threshold);
945  float update_probability_positive = ((float)tm->threshold - (float)votes_clipped_positive) / (float)(2 * tm->threshold);
946 
947  // Apply feedback to: chosen classes - every clause
948  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
949  if (prng_next_float(&(tm->rng)) <= update_probability_positive) {
950  tm_apply_feedback(tm, clause_id, positive_class, 1, X);
951  }
952  }
953 
954  // Continue for negative class
955 negative_feedback:
956 
957  int32_t sum_votes_clipped_negative = 0;
958  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
959  if (!label_arr[class_id]) {
960  sum_votes_clipped_negative += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
961  }
962  }
963  if (sum_votes_clipped_negative == 0) return;
964  int32_t random_vote_negative = prng_next_uint32(&(tm->rng)) % sum_votes_clipped_negative;
965  int32_t accumulated_votes_negative = 0;
966  for (uint32_t class_id = 0; class_id < tm->num_classes; class_id++) {
967  if (!label_arr[class_id]) {
968  accumulated_votes_negative += clip(tm->votes[class_id], (int32_t)tm->threshold) + (int32_t)tm->threshold;
969  if (accumulated_votes_negative >= random_vote_negative) {
970  negative_class = class_id;
971  break;
972  }
973  }
974  }
975 
976  int32_t votes_clipped_negative = clip(tm->votes[negative_class], (int32_t)tm->threshold);
977  float update_probability_negative = ((float)votes_clipped_negative + (float)tm->threshold) / (float)(2 * tm->threshold);
978 
979  for (uint32_t clause_id = 0; clause_id < tm->num_clauses; clause_id++) {
980  if (prng_next_float(&(tm->rng)) <= update_probability_negative) {
981  tm_apply_feedback(tm, clause_id, negative_class, 0, X);
982  }
983  }
984 }
