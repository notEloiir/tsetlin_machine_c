408 static inline void calculate_clause_output(struct StatelessTsetlinMachine *sltm, const uint8_t *X) {
409  // For each clause, check if it is "active" - all necessary literals have the right value
410  for (uint32_t clause_id = 0; clause_id < sltm->num_clauses; clause_id++) {
411  sltm->clause_output[clause_id] = 1;
412  uint8_t empty_clause = 1;
413 
414  // Clause is active if:
415  // - it's not empty (unless skip_empty is unset as should be the case for training)
416  // - each literal present in the clause has the right value (same as the input X)
417  // Iterate over linked list of Tsetlin Automata ids
418  struct TANode *curr_ptr = sltm->ta_state[clause_id];
419  while (curr_ptr != NULL) {
420  empty_clause = 0;
421  if (curr_ptr->ta_id % 2 == X[curr_ptr->ta_id / 2]) {
422  sltm->clause_output[clause_id] = 0;
423  break;
424  }
425  curr_ptr = curr_ptr->next;
426  }
427  if (empty_clause) {
428  sltm->clause_output[clause_id] = 0;
429  }
430  }
431 }

438  memset(sltm->votes, 0, sltm->num_classes*sizeof(int32_t));
